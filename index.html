def buscar_en_sistema(termino, base_de_datos): # Limpiamos los espacios y pasamos a minúsculas para que la búsqueda no falle termino_limpio = termino.strip().lower() if not termino_limpio: return "⚠️ Por favor, ingresa algo para buscar." # Buscamos coincidencias en cualquier parte del texto resultados = [item for item in base_de_datos if termino_limpio in item.lower()] # Mostramos los resultados if resultados: print(f"\n🔍 Resultados para '{termino}': ({len(resultados)} encontrados)") for idx, producto in enumerate(resultados, 1): print(f" {idx}. {producto}") else: print(f"\n❌ No se encontró nada para '{termino}'. ¡Todo el catálogo está disponible pero este término no coincide!") # Tu base de datos sin límites de indexación inventario = [ "Laptop Gamer ASUS Rog", "Monitor Curvo 27 pulgadas", "Cafetera Espresso Delonghi", "Curso Completo de Python 2026", "Teclado Mecánico RGB", "Zapatillas de Running Nike", "Café en grano Premium" ] # --- PRUEBAS DE BUSQUEDA --- # Puedes cambiar lo que está entre comillas para probar diferentes búsquedas: buscar_en_sistema("cafe", inventario) buscar_en_sistema("LAPTOP", inventario) buscar_en_sistema("pantalla", inventario) # Caso donde no encuentra nada
